

# 图的搜索

## 邻接矩阵

图由**顶点（Vertex）** 和**边（Edge）** 组成，邻接矩阵是一个 `n×n` 的二维数组（`n` 是顶点数量），其中：

- 行和列都对应图的顶点（比如 `matrix[i][j]` 表示 “顶点 i” 和 “顶点 j” 的关系）；
- 数组元素的值表示两点间是否有边，或边的权重（根据图的类型而定）。

```cpp
#include <iostream>  // 1. 引入输入输出流头文件
                     // 作用：提供cout/endl等输出功能，用于打印矩阵
#include <vector>    // 2. 引入vector容器头文件
                     // 作用：支持动态二维数组（邻接矩阵）和动态边列表（pair的vector）
#include <climits>   // 3. 引入climits头文件（当前代码未用到，注释说明是为INT_MAX预留）
                     // 备注：代码中实际用0/1表示无边/有边，未用到INT_MAX（带权图才需要）
using namespace std; // 4. 使用std命名空间
                     // 作用：简化代码，无需写std::vector、std::cout、std::pair等前缀
// 5. 注释：说明函数功能——构建无向无权图的邻接矩阵
// 构建无向无权图的邻接矩阵
// 6. 注释：说明参数含义
// n：顶点数量
// edges：边的列表（每对元素表示一条边，如{0,1}表示0和1之间有边）
// 函数返回值：vector<vector<int>> → n×n的邻接矩阵（二维动态数组）
vector<vector<int>> buildAdjMatrix(int n, const vector<pair<int, int>>& edges) {
    // 7. 初始化邻接矩阵：n行n列，所有元素默认值为0（表示“无边”）
    // vector<vector<int>> matrix(n, ...) → 创建n个元素（每行）
    // vector<int>(n, 0) → 每行是一个长度为n的vector，元素全为0
    vector<vector<int>> matrix(n, vector<int>(n, 0));
    // 8. 遍历边列表中的每一条边，更新邻接矩阵
    // const auto& edge：const表示不修改edge，auto自动推导类型（pair<int,int>），&引用传递避免拷贝
    for (const auto& edge : edges) {
        // 9. 提取边的第一个顶点u（比如边{0,1}的u=0）
        int u = edge.first;   
        // 10. 提取边的第二个顶点v（比如边{0,1}的v=1）
        int v = edge.second;  
        // 11. 无向图的核心：边是双向的，因此matrix[u][v]和matrix[v][u]都要设为1（表示“有边”）
        matrix[u][v] = 1;
        matrix[v][u] = 1;
    // 12. 返回构建好的邻接矩阵
    return matrix;
}
// 13. 注释：说明函数功能——打印邻接矩阵
// 打印邻接矩阵
// 参数：const vector<vector<int>>& matrix → const表示不修改矩阵，&引用传递避免拷贝
void printAdjMatrix(const vector<vector<int>>& matrix) {
    // 14. 获取矩阵的行数（即顶点数量n，因为邻接矩阵是n×n的）
    int n = matrix.size();
    // 15. 打印列标题（方便查看：列对应顶点0、1、2、3...）
    cout << "  ";  // 先打印两个空格，对齐行标题（行标题是顶点编号，占1个字符）
    for (int i = 0; i < n; i++) {
        cout << i << " ";  // 打印列编号：0 1 2 3...
    }
    cout << endl;  // 换行，准备打印矩阵内容   
    // 16. 遍历矩阵的每一行（每行对应一个顶点i）
    for (int i = 0; i < n; i++) {
        cout << i << " ";  // 17. 打印行标题（当前行对应的顶点编号i）
        // 18. 遍历当前行的每一列（每列对应顶点j）
        for (int j = 0; j < n; j++) {
            cout << matrix[i][j] << " ";  // 19. 打印matrix[i][j]的值（0/1）
        }
        cout << endl;  // 20. 每行打印完换行
    }
}
// 21. 主函数：程序入口
int main() {
    // 22. 定义顶点数量n=4（顶点编号：0、1、2、3）
    int n = 4;
    // 23. 定义边列表：存储无向边，每对{}表示一条边
    // {0,1} → 顶点0和1之间有边；{0,2} → 0和2有边；{1,3} →1和3有边
    vector<pair<int, int>> edges = {{0,1}, {0,2}, {1,3}};     
    // 24. 调用构建函数，得到邻接矩阵
    vector<vector<int>> adjMatrix = buildAdjMatrix(n, edges);     
    // 25. 打印提示信息，说明接下来输出的是无向无权图的邻接矩阵
    cout << "无向无权图的邻接矩阵：" << endl;
    // 26. 调用打印函数，输出矩阵
    printAdjMatrix(adjMatrix);   
    // 27. 返回0：表示程序正常执行结束（C++主函数规范）
    return 0;
}
```

## 邻接表

### 无向无权图的邻接表

```cpp
#include <iostream>
#include <vector>
#include <utility>
// 构建无向无权图的邻接表
// 6. 注释：说明参数含义
// n：顶点数量
// edges：边列表（每对元素表示一条无向边，如{0,1}表示0和1之间有边）
// 函数返回值：vector<vector<int>> → 邻接表（外层vector对应顶点，内层vector对应邻接顶点）
vector<vector<int>> buildAdjList(int n, const vector<pair<int, int>>& edges) {
    // 7. 初始化邻接表：创建n个空的vector（对应n个顶点，每个顶点初始无邻接顶点）
    // vector<vector<int>> adjList(n) → 外层vector有n个元素（索引0~n-1对应顶点0~n-1）
    // 每个元素是一个空的vector<int> → 初始时每个顶点的邻接列表为空
    vector<vector<int>> adjList(n);
    // 8. 遍历边列表中的每一条边，为邻接表添加邻接顶点
    // const auto& edge：
    // - const：保证遍历过程中不修改edge的值，提升代码安全性；
    // - auto：自动推导edge的类型为pair<int, int>，避免手动写冗长类型；
    // - &：引用传递，避免拷贝edge（边数量多时可节省内存和时间）；
    // - range-for循环：遍历edges中的每一个pair元素（每一条边）
    for (const auto& edge : edges) {
        // 9. 提取当前边的第一个顶点u（比如边{0,1}的u=0）
        int u = edge.first;
        // 10. 提取当前边的第二个顶点v（比如边{0,1}的v=1）
        int v = edge.second;
        // 11. 无向图核心操作1：将v添加到u的邻接列表中（表示u和v相邻）
        adjList[u].push_back(v);  
        // 12. 无向图核心操作2：将u添加到v的邻接列表中（表示v和u相邻，体现无向边的双向性）
        adjList[v].push_back(u);  
    }  
    // 13. 返回构建完成的邻接表
    return adjList;
}
// 14. 注释：说明函数功能——打印邻接表（直观展示每个顶点的邻接顶点）
// 打印邻接表（直观展示每个顶点的邻接顶点）
// 参数：const vector<vector<int>>& adjList → 
// - const：保证打印过程中不修改邻接表；
// - &：引用传递，避免拷贝整个邻接表（尤其顶点/边数量多时，拷贝成本高）
void printAdjList(const vector<vector<int>>& adjList) {
    // 15. 获取邻接表的顶点数量（外层vector的大小就是顶点数n）
    int n = adjList.size();
    // 16. 遍历每一个顶点（索引i对应顶点i）
    for (int i = 0; i < n; i++) {
        // 17. 打印当前顶点的提示信息，方便阅读
        cout << "顶点" << i << "的邻接顶点：";       
        // 18. 遍历顶点i的邻接列表（内层vector），打印所有邻接顶点
        // range-for循环：遍历adjList[i]中的每一个int元素（邻接顶点编号）
        for (int v : adjList[i]) {
            cout << v << " ";  // 19. 打印邻接顶点编号，空格分隔
        }       
        cout << endl;  // 20. 每个顶点的邻接列表打印完后换行，格式更清晰
    }
}
// 21. 主函数：程序入口
int main() {
    // 22. 定义顶点数量n=4（顶点编号：0、1、2、3）
    int n = 4;
    // 23. 定义边列表：存储无向边，每对{}表示一条边
    // {0,1} → 顶点0和1之间有边；{0,2} → 0和2有边；{1,3} → 1和3有边
    vector<pair<int, int>> edges = {{0,1}, {0,2}, {1,3}};        
    // 24. 调用构建函数，得到邻接表
    vector<vector<int>> adjList = buildAdjList(n, edges);      
    // 25. 打印提示信息，说明接下来输出的是无向无权图的邻接表
    cout << "无向无权图的邻接表：" << endl;
    // 26. 调用打印函数，输出邻接表
    printAdjList(adjList);       
    // 27. 返回0：表示程序正常执行结束（C++主函数规范）
    return 0;
}
```

### 带权图的邻接表实现

```cpp
#include <iostream>
#include <vector>
#include <tuple>   // 用于存储(顶点u, 顶点v, 权重)
#include <utility> // 用于pair
// 6. 注释：说明函数功能——构建无向带权图的邻接表
// 构建无向带权图的邻接表
// 函数返回值：vector<vector<pair<int, int>>> → 
// - 外层vector：索引对应顶点编号（0~n-1）；
// - 内层vector<pair<int, int>>：每个元素是pair，first=邻接顶点，second=边的权重
// 参数：
// - n：顶点数量；
// - edges：带权边列表，每个元素是tuple<int, int, int>（u, v, weight）
vector<vector<pair<int, int>>> buildWeightedAdjList(int n, const vector<tuple<int, int, int>>& edges) {
    // 7. 初始化邻接表：n个顶点，每个顶点对应一个空的vector<pair<int, int>>
    // 初始时每个顶点无邻接顶点，邻接列表为空
    vector<vector<pair<int, int>>> adjList(n);
    
    // 8. 遍历所有带权边，为邻接表添加“邻接顶点+权重”
    // const auto& edge：const保证不修改edge，auto推导类型为tuple<int,int,int>，&引用传递避免拷贝
    for (const auto& edge : edges) {
        // 9. 提取tuple中第0个元素（顶点u）→ 带权边的第一个顶点
        int u = get<0>(edge);    
        // 10. 提取tuple中第1个元素（顶点v）→ 带权边的第二个顶点
        int v = get<1>(edge);    
        // 11. 提取tuple中第2个元素（weight）→ 边u-v的权重
        int weight = get<2>(edge);

        // 12. 无向带权图核心操作1：向顶点u的邻接表添加（v, weight）
        // emplace_back：直接在vector中构造pair对象（无需先创建临时pair），比push_back更高效
        adjList[u].emplace_back(v, weight);
        // 13. 无向带权图核心操作2：向顶点v的邻接表添加（u, weight）（体现无向边的双向性）
        adjList[v].emplace_back(u, weight);
    }

    // 14. 返回构建完成的带权邻接表
    return adjList;
}

// 15. 注释：说明函数功能——打印带权邻接表
// 打印带权邻接表
// 参数：const vector<vector<pair<int, int>>>& adjList → const保证不修改，&引用传递避免拷贝
void printWeightedAdjList(const vector<vector<pair<int, int>>>& adjList) {
    // 16. 获取顶点数量（外层vector的大小）
    int n = adjList.size();

    // 17. 遍历每一个顶点（索引i对应顶点i）
    for (int i = 0; i < n; i++) {
        // 18. 打印当前顶点的提示信息，明确格式（顶点:权重）
        cout << "顶点" << i << "的邻接顶点（顶点:权重）：";
        
        // 19. 遍历顶点i的邻接列表（每个元素是pair<int, int>）
        // const auto& p：const保证不修改，auto推导为pair<int,int>，&引用传递避免拷贝
        for (const auto& p : adjList[i]) {
            // 20. 打印“邻接顶点:权重”，比如p.first=1、p.second=5 → 输出“1:5”
            cout << p.first << ":" << p.second << " ";
        }
        
        cout << endl;  // 21. 每个顶点打印完换行，格式更清晰
    }
}

// 22. 主函数：程序入口
int main() {
    // 23. 定义顶点数量n=4（顶点编号：0、1、2、3）
    int n = 4;
    // 24. 定义带权边列表：每个tuple表示（u, v, weight）
    // {0,1,5} → 顶点0和1之间的边权重为5；
    // {0,2,3} → 顶点0和2之间的边权重为3；
    // {1,3,2} → 顶点1和3之间的边权重为2；
    vector<tuple<int, int, int>> edges = {{0,1,5}, {0,2,3}, {1,3,2}};
    
    // 25. 调用构建函数，得到带权邻接表
    // auto：自动推导adjList的类型为vector<vector<pair<int, int>>>，简化代码
    auto adjList = buildWeightedAdjList(n, edges);
    
    // 26. 打印提示信息
    cout << "无向带权图的邻接表：" << endl;
    // 27. 调用打印函数，输出带权邻接表
    printWeightedAdjList(adjList);
    
    // 28. 返回0：程序正常结束
    return 0;
}
```

## 广度优先搜索

### 基于邻接表的 BFS

```cpp
#include <iostream>  // 1. 引入输入输出流头文件，用于cout打印遍历顺序
#include <vector>    // 2. 引入vector容器头文件，支持邻接表（二维vector）和visited数组
#include <queue>     // 3. 引入queue头文件（BFS核心！），提供队列的push/pop/front/empty等操作
#include <utility>   // 4. 引入utility头文件，支持pair（边列表的存储）
using namespace std;  // 5. 使用std命名空间，简化cout/vector/queue/pair等调用
// 6. 注释：复用之前的邻接表构建函数（无向无权图）
// 复用之前的邻接表构建函数（无向无权图）
// 函数作用：根据顶点数n和边列表，构建无向无权图的邻接表
// 参数：n=顶点数，edges=边列表（pair<int,int>表示每条无向边）
// 返回值：vector<vector<int>> 类型的邻接表
vector<vector<int>> buildAdjList(int n, const vector<pair<int, int>>& edges) {
    // 7. 初始化邻接表：n个顶点，每个顶点对应空的vector（初始无邻接顶点）
    vector<vector<int>> adjList(n);    
    // 8. 遍历边列表，为无向图双向添加邻接顶点
    // const auto& edge：const防修改、auto推导类型为pair<int,int>、&引用传递避免拷贝
    for (const auto& edge : edges) {
        int u = edge.first;   // 9. 提取边的第一个顶点u
        int v = edge.second;  // 10. 提取边的第二个顶点v       
        // 11. 无向图核心：双向添加邻接顶点（体现边的双向性）
        adjList[u].push_back(v);
        adjList[v].push_back(u);
    }    
    // 12. 返回构建完成的邻接表
    return adjList;
}
// 13. 注释：定义广度优先搜索（BFS）函数
// 广度优先搜索（BFS）
// 函数参数说明：
// - adjList：const引用的邻接表（不修改，避免拷贝）；
// - start：BFS的起始顶点；
// - visited：引用传递的bool数组（标记顶点访问状态，方便外部查看/复用）
void bfs(const vector<vector<int>>& adjList, int start, vector<bool>& visited) {
    // 14. 初始化队列：存储“待访问的顶点”，队列元素类型为int（顶点编号）
    // queue<int>是C++标准库队列，遵循“先进先出（FIFO）”，是BFS的核心数据结构
    queue<int> q;
    // 15. 起始顶点入队：将start加入队列尾部，等待处理
    q.push(start);
    // 16. 标记起始顶点为已访问：避免后续重复入队（关键！防止无向图回路导致无限循环）
    visited[start] = true;
    // 17. 打印提示信息，说明BFS的起始顶点
    cout << "BFS遍历顺序（起始顶点" << start << "）：";
    // 18. BFS核心循环：只要队列不为空，就持续处理待访问顶点
    // 队列空 → 所有可达顶点已处理完毕
    while (!q.empty()) {
        // 19. 取出队首顶点：获取当前要处理的顶点（队列头部元素）
        int curr = q.front();
        // 20. 队首顶点出队：从队列中移除已取出的顶点（必须执行，否则队列永远不为空）
        q.pop();  
        // 21. 访问当前顶点：此处仅打印（可替换为任意业务逻辑，如统计、路径记录等）
        cout << curr << " ";
        // 22. 遍历当前顶点curr的所有邻接顶点
        // 邻接表中adjList[curr]是curr的所有邻接顶点列表
        for (int neighbor : adjList[curr]) {
            // 23. 仅处理“未访问”的邻接顶点（核心过滤条件）
            if (!visited[neighbor]) {
                q.push(neighbor);          // 24. 未访问的邻接顶点入队，等待后续处理
                visited[neighbor] = true;  // 25. 标记为已访问（入队时标记，而非出队时！）
            }
        }
    }
    // 26. 遍历完成后换行，优化输出格式
    cout << endl;
}
// 27. 主函数：程序入口
int main() {
    // 28. 定义顶点数量n=4（顶点编号：0、1、2、3）
    int n = 4;
    // 29. 定义边列表：{0,1}（0-1边）、{0,2}（0-2边）、{1,3}（1-3边）
    vector<pair<int, int>> edges = {{0,1}, {0,2}, {1,3}};
    // 30. 构建邻接表：调用buildAdjList函数，得到无向无权图的邻接表
    vector<vector<int>> adjList = buildAdjList(n, edges);
    // 31. 初始化访问标记数组：大小为n，初始值全为false（所有顶点未访问）
    vector<bool> visited(n, false);
    // 32. 执行BFS：起始顶点为0，传入邻接表和visited数组
    bfs(adjList, 0, visited);
    // 33. 返回0：程序正常结束
    return 0;
}
```

### BFS 找最短路径（拓展）

<span style="font-weight:bold; font-style:italic; font-size:1.1em;">BFS 最经典的应用是 “无权图最短路径”，只需在 BFS 基础上记录每个顶点的 “前驱顶点”，最终回溯得到最短路径</span>

```cpp
#include <iostream>
#include <vector>
#include <queue>
#include <utility>
#include <algorithm>  // 用于reverse反转路径
using namespace std;
// 构建无向无权图的邻接表
vector<vector<int>> buildAdjList(int n, const vector<pair<int, int>>& edges) {//  容器类型：二维vector（外层存顶点，内层存邻接顶点）
    vector<vector<int>> adjList(n);
    for (const auto& edge : edges) {
        int u = edge.first;
        int v = edge.second;
        adjList[u].push_back(v);
        adjList[v].push_back(u);
    }
    return adjList;
}
// 函数功能：BFS找无权图中从start到target的最短路径
// 参数：adjList=邻接表，start=起始顶点，target=目标顶点
void bfsShortestPath(const vector<vector<int>>& adjList, int start, int target) {
    // 1. 访问标记数组：避免顶点重复入队，和基础BFS作用一致
    vector<bool> visited(adjList.size(), false);   
    // 2. 前驱数组prev：核心！prev[i]表示“顶点i是从哪个顶点过来的”
    // 初始值-1：表示顶点i无前置顶点（起始顶点的prev始终是-1，未访问顶点也为-1）
    vector<int> prev(adjList.size(), -1);      
    // 3. BFS队列：存储待处理的顶点，保证层序遍历
    queue<int> q;
    // 4. 起始顶点入队+标记访问：BFS的标准初始化
    q.push(start);
    visited[start] = true;
    // 5. BFS主循环：处理队列中的顶点，直到队空或找到目标
    while (!q.empty()) {
        int curr = q.front();
        q.pop();
        // 6. 提前退出优化：找到目标顶点后，无需继续遍历其他顶点，节省时间
        if (curr == target) break;  
        // 7. 遍历当前顶点的所有邻接顶点
        for (int neighbor : adjList[curr]) {
            if (!visited[neighbor]) {
                q.push(neighbor);
                visited[neighbor] = true;
                // 8. 关键操作：记录neighbor的前驱是curr
                // 比如neighbor=1是从curr=0来的 → prev[1]=0；
                // neighbor=3是从curr=1来的 → prev[3]=1；
                prev[neighbor] = curr;  
            }
        }
    }
    // 9. 回溯前驱数组构建路径：从target倒推到start
    cout << "从" << start << "到" << target << "的最短路径：";
    vector<int> path;
    // 循环逻辑：i初始为target，每次取prev[i]作为下一个顶点，直到i=-1（start的prev是-1）
    // 示例（0→3）：i=3 → prev[3]=1 → i=1 → prev[1]=0 → i=0 → prev[0]=-1 → 循环结束
    // 此时path = [3,1,0]（倒序路径）
    for (int i = target; i != -1; i = prev[i]) {
        path.push_back(i);
    }    
    // 10. 反转路径：将[3,1,0]转为[0,1,3]（正序最短路径）
    reverse(path.begin(), path.end());  
    // 11. 打印正序路径
    for (int num : path) {
        cout << num << " ";
    }
    cout << endl;
}
int main() {
    int n = 4;
    vector<pair<int, int>> edges = {{0,1}, {0,2}, {1,3}};
    vector<vector<int>> adjList = buildAdjList(n, edges);
    // 找从0到3的最短路径
    bfsShortestPath(adjList, 0, 3);
    return 0;
}
```

## 贝尔曼 - 福特算法

```cpp
#include <iostream>
#include <vector>
#include <climits>  // 仅用于参考INT_MAX，实际不用它
using namespace std;
// 定义边的结构体：u=起点，v=终点，weight=权重
struct Edge {
    int u;//起点
    int v;//终点
    int weight;//权重
    // 构造函数：方便初始化边
    Edge(int u_, int v_, int weight_) : u(u_), v(v_), weight(weight_) {}//初始化列表：用括号里的参数值，给结构体的成员变量 “赋初始值”
};
// 贝尔曼-福特算法核心函数
// 参数：edges-所有边的集合，vertex_num-顶点数量，start-源点（顶点索引，从0开始）
// 返回值：存储源点到各顶点最短距离的数组（若有负权环则返回空vector）
vector<int> bellmanFord(vector<Edge>& edges, int vertex_num, int start) {
    // 1. 初始化距离数组：无穷大（用1e9，避免INT_MAX加法溢出），源点距离为0
    const int INF = 1e9;//是贝尔曼 - 福特算法（以及所有最短路径算法）中规避数值问题的关键技巧
    vector<int> dist(vertex_num, INF);//vertex_num（顶点数）
    dist[start] = 0;  // 起点到自身的距离为0
    // 2. 核心：迭代 V-1 次，对所有边执行松弛操作
    for (int i = 0; i < vertex_num - 1; ++i) {
        bool updated = false;  // 优化：标记是否有距离更新，无更新则提前退出
        // 遍历所有边
        for (Edge& e : edges) {
            int u = e.u;
            int v = e.v;
            int w = e.weight;
            // 松弛条件：
            // 1. dist[u] != INF → 确保源点能到达u（否则u→v的路径无意义）
            // 2. dist[u] + w < dist[v] → 走u→v能让到v的距离更短
            if (dist[u] != INF && dist[u] + w < dist[v]) {
                dist[v] = dist[u] + w;
                updated = true;
            }
        }
        // 本次迭代无更新，说明最短路径已全部找到，提前结束
        if (!updated) break;
    }
    // 3. 检测负权环：第V次迭代，若仍能松弛则存在负权环
    for (Edge& e : edges) {
        int u = e.u;
        int v = e.v;
        int w = e.weight;
        if (dist[u] != INF && dist[u] + w < dist[v]) {
            // 存在负权环，返回空数组
            cout << "图中存在负权环，无最短路径！" << endl;
            return vector<int>();
        }
    }
    // 无负权环，返回最短距离数组
    return dist;
}
// 主函数：测试算法
int main() {
    // 定义图的参数：顶点数=5（索引0-4），边集合
    int vertex_num = 5;
    vector<Edge> edges;
    // 按Python版的测试用例添加边：(u, v, weight)
    edges.emplace_back(0, 1, 6);
    edges.emplace_back(0, 2, 7);
    edges.emplace_back(1, 3, 5);
    edges.emplace_back(1, 4, -4);
    edges.emplace_back(2, 1, -2);
    edges.emplace_back(3, 1, -2);
    edges.emplace_back(3, 4, 3);
    edges.emplace_back(4, 3, 1);
    // 源点设为0
    int start = 0;
    // 调用贝尔曼-福特算法
    vector<int> shortest_dist = bellmanFord(edges, vertex_num, start);
    // 输出结果
    if (!shortest_dist.empty()) {
        cout << "从源点 " << start << " 到各顶点的最短路径：" << endl;
        for (int i = 0; i < vertex_num; ++i) {
            cout << "到顶点 " << i << " 的最短距离：" << shortest_dist[i] << endl;
        }
    }
    return 0;
}
```

## 狄克斯特拉算法

### 核心概述

#### 定义：

 用于求解 **带权图中，从一个起点（源点）到其他所有节点的最短路径** 的贪心算法。

#### 硬性前提

**图中所有边的权值 ≥ 0**（负权边会直接导致算法失效）。

#### 核心思想（贪心策略）

每次选择 **当前离起点最近、且未确定最终最短路径** 的节点；

用这个节点去**松弛**它的所有邻边（更新邻居节点的最短距离）；

标记该节点为「已确定最短路径」，重复直到所有节点处理完毕。

#### 时间复杂度

- 暴力版：O(N2)（适合小图）
- 堆优化版：O(MlogN)（适合大图，竞赛 / 工程首选）

> N= 节点数，M= 边数

### 核心概念

#### 图的存储：邻接表（STL 标准实现）

不用低效的邻接矩阵，用 `vector` 存储邻接表（最常用）：

```cpp
//邻接表定义：adj[u] 存储{v,w}->从u到v有一条权值为w的边
vector<vector<pair<int,int>>>adj;
```

#### 核心变量

`dist[]`：距离数组

- `dist[i]` = 起点到节点 `i` 的**当前最短距离**
- 初始化：起点 = 0，其余 = 无穷大（`INT_MAX`）

`vis[]`：标记数组

- `vis[i] = true`：节点 `i` 的最短路径**已最终确定**，不再修改

**松弛操作**（算法灵魂）

对边 `u → v`（权值 w）：

```cpp
if(dist[v]>dist[u]+w){
    dist[v]=dist[u]+w;
}
```

含义：如果「经过 u 到 v」比「直接到 v」更近，就更新 v 的最短距离。

### 执行步骤

初始化：`dist[起点] = 0`，其余 `dist[] = ∞`，`vis[] = false`；

循环 N 次（N 为节点数）：

① 找到 **未被访问** 且 `dist` 最小的节点 `u`；

② 标记 `vis[u] = true`（确定 u 的最短路径）；

③ 遍历 u 的所有邻边 `u→v`，执行**松弛操作**；

最终 `dist[]` 数组就是起点到所有节点的最短路径。

### 实现

#### 暴力实现

适合**小数据量**，每次遍历找最小距离节点。

```cpp
#include <iostream>
#include <vector>
#include <climits>
using namespace std;
//暴力版 Dijkstra: n=节点数，start=起点
void dijkstra_brute(int n,int start,vector<vector<pair<int,int>>>% adj){
	//1.初始化距离数组和标记数组
    vector<int> dist)n,INT_MAX+);
    vector<int> vis(n,false);
    dist[start]=0;//起点距离为0
    //2.循环n次，处理所有节点
    for(int i=0;i<n;++i){
        //找 未访问 + dist 最小的节点u
        int u=-1;
        int min_dist=INT_MAX;
        for(int j=0;j<n;++j){
			if(!vis[j]&&dist[j]<min_dist){
				min_dist=dist[j];
                u=j;
            }
        }
        if(u==-1) break;//剩余节点不可达，推出
        //标记u已确定最短路径
        vis[u]=true;
        //松弛所有临边
        for(auto&edge:adj[u]){
            int v=edge.first;//邻接节点
            int w=edge/second;//边权
            //松弛操作(防止溢出:dist[u]!=INT_MAX)
            if(dist[u]!=INT_MAX&&dist[v]>dist[u]+w){
				dist[v]=dist[u]+w;
            }
        }
    }
    //输出结果
    cout<<"起点"<< start << " 到各节点的最短距离：\n";
    for(int i=0;i<n;++i){
		if(dist[i]==INT_MAX)cout<<i<<"不可达"<<endl;
        else cout<<i<<":"<<dist[i]<<endl;
    }
}

    int n=4;
    vector<vector<pair<int,int>>>age(n);
    //添加边：u->v,w
    adj[0]。emplace_back(1,2);
    adj[0].emplace_back(2, 5);
    adj[1].emplace_back(2, 1);
    adj[1].emplace_back(3, 3);
    adj[2].emplace_back(3, 1);
    // 起点0，运行算法
    dijkstra_brute(n, 0, adj);
    return 0;    
}
```

#### 堆优化实现

用 **优先队列（小根堆）** 自动找最小距离节点，效率大幅提升。

```cpp
#include <iostream>
#include <vector>//引用向量容器，用于存储邻接表和距离数组
#include <queue>//引入队列库
#include <climits>//引入整数极限常量
#include <utility>//引入实用工具库
using namespace std;
//堆优化版 Dijkstra
void dijkstra_heap(int n,int start,vector<vector<pair<int,int>>>& adj){
    vector<int> dist(n,INT_MAX);
    dist[start]=0;
    //小根堆：存储 {距离, 节点}，默认大根堆，用 greater<> 改为小根堆
    priority_queue<pair<int,int>,
    	vector<pair<int,int>>,
   	 	greater<pair<int,int>>> pq;
    pq.emplace(0,start);//入队：起点距离0
    while(!pq.empty()){
        //取出堆顶：最小距离节点
        auto [d,u]=pq.top();
        pq.pop();
        //优化：如果当前距离》已纪录的最短距离，直接跳过
        if(d>dist[u])continue;
        //松弛临边
        for(auto& edge:adj[u]){
            int v=edge.first;
            int w=edge.second;
            if(dist[u]!=INT_MAX&&dist[v]>dist[u]+w){
                dist[u]=dist[u]+w;
                pq.emplace(dist[v],v);//更新后入队
            }
        }
    }
     // 输出结果
    cout << "\n堆优化版结果：\n";
    for (int i = 0; i < n; ++i) {
        if (dist[i] == INT_MAX) cout << i << ": 不可达" << endl;
        else cout << i << ": " << dist[i] << endl;
    }
}
int main() {
    int n = 4;
    vector<vector<pair<int, int>>> adj(n);
    adj[0].emplace_back(1, 2);
    adj[0].emplace_back(2, 5);
    adj[1].emplace_back(2, 1);
    adj[1].emplace_back(3, 3);
    adj[2].emplace_back(3, 1);
    dijkstra_heap(n, 0, adj);
    return 0;
}
```

## 关键注意事项

**边权必须非负**

负权边会破坏贪心策略，导致算法失效（负权用 Bellman-Ford / SPFA）。

**堆优化的去重**

优先队列中可能存在同一个节点的多个距离记录，用 `if (d > dist[u]) skip` 优化。

**无穷大的判断**

必须判断 `dist[u] != INT_MAX`，否则会整数溢出。

**适用场景**

导航软件、网络路由、任务调度等**带权非负图**的最短路径求解。
