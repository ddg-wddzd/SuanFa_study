

# 数组的查找



## 线性查找

```cpp
#include <iostream>
#include <vector>
using namespace std;
//返回下标
int linearSearch1(const vector<int>& arr, int target) {
    for (int i = 0; i < arr.size(); i++) {
        if (arr[i] == target)
            return i;   // 找到返回下标
    }
    return -1;  // 没找到
}
//返回是否存在
bool linearSearch2(const vector<int>& arr, int target) {
    for (int num : arr) {
        if (num == target)
            return true;
    }
    return false;
}
```



## 二分查找

<span style="font-weight:bold; font-style:italic; text-decoration:underline; font-size:1.3em;"> 标准模板</span>

```cpp
// 定义二分查找函数
// int：返回值为目标值的索引（找到）或-1（未找到）
// vector<int>& arr：待查找的有序数组（引用传递，避免拷贝，且数组需是升序排列）
// int target：要查找的目标值
int binarySearch(vector<int>& arr, int target) {
    // 1. 初始化左边界：查找范围的起始索引（数组第一个元素）
    int left = 0;
    // 2. 初始化右边界：查找范围的结束索引（数组最后一个元素）
    int right = arr.size() - 1;
    // 3. 核心循环：只要查找范围有效（左边界≤右边界），就继续查找
    // 循环条件left <= right：保证最后一个元素（left=right时）也会被检查
    while (left <= right) {
        // 4. 计算中间位置mid：避免溢出的安全写法
        // 写法：left + (right - left)/2 而非 (left+right)/2
        // 原因：若left和right都是大数（如2^30），left+right会超出int取值范围，导致溢出
        // 等价于 (left+right)/2 的整数除法结果，但更安全
        int mid = left + (right - left) / 2;
        // 5. 情况1：中间元素等于目标值 → 找到，直接返回mid（目标值的索引）
        if (arr[mid] == target)
            return mid;
        // 6. 情况2：中间元素小于目标值 → 目标值在右半区间，调整左边界
        else if (arr[mid] < target)
            left = mid + 1;  // 关键：mid已检查过，无需再包含，直接+1
        // 7. 情况3：中间元素大于目标值 → 目标值在左半区间，调整右边界
        else
            right = mid - 1; // 关键：mid已检查过，无需再包含，直接-1
    }
    // 8. 循环结束仍未找到 → 返回-1（约定俗成的“未找到”标识）
    return -1;
}
```

### 查找等于 target

```cpp
int binarySearch(vector<int>& arr, int target) {
    int left = 0, right = arr.size() - 1;
    while (left <= right) {
        int mid = left + (right - left) / 2;
        if (arr[mid] == target)
            return mid;
        else if (arr[mid] < target)
            left = mid + 1;
        else
            right = mid - 1;
    }
    return -1;
}
```

### 查找第一个 ≥ target（lower_bound）

```cpp
int lowerBound(vector<int>& arr, int target) {
    int left = 0, right = arr.size() - 1;
    while (left <= right) {
        int mid = left + (right - left) / 2;
        if (arr[mid] >= target)
            right = mid - 1;
        else
            left = mid + 1;
    }
    return left;
}
```
